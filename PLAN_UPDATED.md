# Medialogg - Unified Media Logging Platform
# Updated Plan: SvelteKit + Go Stack

## Executive Summary

Medialogg is a unified platform for logging, reviewing, and sharing media consumption across multiple formats (films, anime, books, manga, games, doujin). 

**Tech Stack: SvelteKit (Frontend) + Go/Fiber (Backend) + PostgreSQL**

---

## 1. Architecture Overview

```
┌─────────────────┐      HTTP/REST       ┌──────────────────┐
│   SvelteKit     │ ◄──────────────────► │   Go (Fiber)     │
│   (Frontend)    │                      │   (API Server)   │
│   Port 3000     │                      │   Port 8080      │
└─────────────────┘                      └────────┬─────────┘
                                                  │
                                                  ▼
                                         ┌──────────────────┐
                                         │   PostgreSQL     │
                                         │   Port 5432      │
                                         └──────────────────┘
```

### Why This Stack?

| Component | Choice | Rationale |
|-----------|--------|-----------|
| **Frontend** | SvelteKit | Fastest bundle size, minimal JS, Vite-powered, self-host friendly |
| **Backend** | Go + Fiber | Single binary, low memory, high concurrency, fast |
| **Database** | PostgreSQL | Reliable, great JSON support, self-hostable |
| **API** | REST + OpenAPI | Simple, language-agnostic, easy to document |

```typescript
// src/lib/api/client.ts
const API_BASE = import.meta.env.VITE_API_URL || 'http://localhost:8080';

export async function api<T>(
    endpoint: string,
    options: RequestInit = {}
): Promise<T> {
    const token = localStorage.getItem('token');
    
    const res = await fetch(`${API_BASE}/api${endpoint}`, {
        ...options,
        headers: {
            'Content-Type': 'application/json',
            ...(token && { Authorization: `Bearer ${token}` }),
            ...options.headers
        }
    });
    
    if (!res.ok) {
        const error = await res.json().catch(() => ({ message: 'Unknown error' }));
        throw new Error(error.message);
    }
    
    return res.json();
}

// Auth
export const auth = {
    register: (data: RegisterData) => api('/auth/register', {
        method: 'POST',
        body: JSON.stringify(data)
    }),
    login: (data: LoginData) => api('/auth/login', {
        method: 'POST',
        body: JSON.stringify(data)
    })
};

// Logs
export const logs = {
    getTimeline: () => api<Log[]>('/logs/timeline'),
    getMyLogs: () => api<Log[]>('/logs/me'),
    create: (data: CreateLogData) => api<Log>('/logs', {
        method: 'POST',
        body: JSON.stringify(data)
    }),
    update: (id: string, data: UpdateLogData) => api<Log>(`/logs/${id}`, {
        method: 'PUT',
        body: JSON.stringify(data)
    })
};

// Media
export const media = {
    search: (query: string) => api<Media[]>(`/media?search=${encodeURIComponent(query)}`),
    get: (id: string) => api<Media>(`/media/${id}`)
};
```

### Store Setup

```typescript
// src/lib/stores/auth.ts
import { writable } from 'svelte/store';
import { browser } from '$app/environment';

interface User {
    id: string;
    username: string;
    displayName?: string;
    avatar?: string;
}

function createAuthStore() {
    const { subscribe, set } = writable<User | null>(null);
    
    return {
        subscribe,
        login: (user: User, token: string) => {
            if (browser) {
                localStorage.setItem('token', token);
                localStorage.setItem('user', JSON.stringify(user));
            }
            set(user);
        },
        logout: () => {
            if (browser) {
                localStorage.removeItem('token');
                localStorage.removeItem('user');
            }
            set(null);
        },
        init: () => {
            if (browser) {
                const user = localStorage.getItem('user');
                if (user) set(JSON.parse(user));
            }
        }
    };
}

export const auth = createAuthStore();
```

```yaml
openapi: 3.0.0
info:
  title: Medialogg API
  version: 1.0.0

paths:
  /auth/register:

      summary: Register new user

        required: true
        content:
          application/json:

              type: object
              properties:
                username: { type: string }
                email: { type: string }

        201:
          description: User created
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/User'

  /logs:
    post:
      summary: Create activity log
      security:
        - bearerAuth: []
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              properties:
                mediaId: { type: string }
                status: { type: string, enum: [planned, in_progress, completed, dropped] }
                rating: { type: number, minimum: 0, maximum: 10 }
                note: { type: string }
      responses:
        201:
          description: Log created

components:
  schemas:

      type: object
      properties:
        id: { type: string }
        username: { type: string }
        displayName: { type: string }
        avatar: { type: string }
    
    Log:
      type: object
      properties:

        status: { type: string }

        note: { type: string }

  securitySchemes:
    bearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT
```
